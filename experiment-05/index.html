productSchema.methods.updateStock = function (sku, quantity) {
  const variant = this.variants.find((item) => item.sku === sku);
  if (!variant || variant.stock + quantity < 0) throw new Error("Invalid stock update");
  variant.stock += quantity;
  return this.save();
};
